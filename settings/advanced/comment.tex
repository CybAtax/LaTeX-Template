\usepackage{blindtext}
\usepackage{xcolor}
\usepackage{pdfcomment}
\usepackage{datetime}

% Text-to-highlight comment
\NewDocumentCommand{\todo}{O{} m}{
    \ifthenelse{\isempty{#1}}
        {\pdfcomment[icon=Note]{#2}}
        {\pdfmarkupcomment[markup=Highlight, icon=Note, color=yellow]{#1}{#2}}
    \PackageWarning{Internal}{#1}
}

\NewDocumentCommand{\fixme}{O{} m}{
    % Error needs to be inside condition or else markup comment will not render
    \ifthenelse{\isempty{#1}}
        {\pdfcomment[icon=Note, color=red]{#2} \PackageError{Internal}{#2}{}}
        {\pdfmarkupcomment[markup=Highlight, icon=Note, color=red]{#1}{#2}\PackageError{Internal}{#2}{}}
}
