\usepackage{blindtext}
\usepackage{pdfcomment}
\usepackage{datetime}

% Text-to-highlight comment color
\NewDocumentCommand{\todo}{O{} m O{yellow}}{
    \ifthenelse{\isempty{#1}}
        {\pdfcomment[icon=Note,color=#3]{#2} \hspace{0.5cm} }
        {\pdfmarkupcomment[markup=Highlight, icon=Note, color=#3]{#1}{#2}}
    \PackageWarning{Internal}{#2}
}

% Text-to-highlight comment color
\NewDocumentCommand{\fixme}{O{} m O{red}}{
    % Error needs to be inside condition or else markup comment will not render
    \ifthenelse{\isempty{#1}}
        {\pdfcomment[icon=Note, color=#3]{#2} \hspace{0.5cm} \PackageError{Internal}{#2}{}}
        {\pdfmarkupcomment[markup=Highlight, icon=Note, color=#3]{#1}{#2}\PackageError{Internal}{#2}{}}
}
