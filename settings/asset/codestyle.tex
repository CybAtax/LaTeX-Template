% language definition: Java (example)
\lstdefinestyle{java}{
  otherkeywords={var},
  literate=
  % Curly braces
  {\{}{{\CodeSymbol{\{}}}1
  {\}}{{\CodeSymbol{\}}}}1
  % Parantheses
  {(}{{\CodeSymbol{(}}}1
  {)}{{\CodeSymbol{)}}}1
  % Brackets
  {[}{{\CodeSymbol{[}}}1
  {]}{{\CodeSymbol{]}}}1
  % Maths
  {>}{{\CodeSymbol{$>$}}}1
  {=}{{\CodeSymbol{$=$}}}1
  {;}{{\CodeSymbol{$;$}}}1
  {@}{{\textcolor{pred}{@}}}1
}

}
